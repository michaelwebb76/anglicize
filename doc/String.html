def copy_case(word)
  result = ""
  case word
  when word.upcase
    result = self.upcase
  when word.downcase
    result = self.downcase
  else
    self.chars.each_with_index do |char, index|
      word_char = word[index] || "a"
      result << (word_char.upcase == word_char ? char.upcase : char.downcase)
    end
  end
  result
end
